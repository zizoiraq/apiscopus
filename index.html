const fetch = require('node-fetch'); exports.handler = async (event) => { // السماح بـ CORS const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type', 'Content-Type': 'application/json' }; if (event.httpMethod === 'OPTIONS') { return { statusCode: 204, headers }; } try { const { apiKey, searchTerm } = JSON.parse(event.body); if (!apiKey || !searchTerm) { return { statusCode: 400, headers, body: JSON.stringify({ error: 'API Key ومصطلح البحث مطلوبان' }) }; } // تحديد نوع البحث (ISSN أو اسم) let query = ''; if (/^\d{4}-\d{3}[\dX]$/i.test(searchTerm) || /^\d{7,8}$/.test(searchTerm)) { query = `ISSN(${searchTerm})`; } else { query = `SRCTITLE(${encodeURIComponent(searchTerm)})`; } // استخدام Scopus Serial Title API const url = `https://api.elsevier.com/content/serial/title?query=${query}&view=STANDARD`; const response = await fetch(url, { headers: { 'Accept': 'application/json', 'X-ELS-APIKey': apiKey } }); if (!response.ok) { return { statusCode: response.status, headers, body: JSON.stringify({ error: `خطأ ${response.status}: ${response.statusText}` }) }; } const data = await response.json(); const entries = data['serial-metadata-response']?.entry; if (!entries || (Array.isArray(entries) && entries.length === 0)) { return { statusCode: 200, headers, body: JSON.stringify({ found: false }) }; } const journal = Array.isArray(entries) ? entries[0] : entries; // استخراج CiteScore let citeScore = ''; if (journal['citeScoreYearInfoList']) { citeScore = journal['citeScoreYearInfoList']['citeScoreCurrentMetric'] || ''; } const result = { found: true, journal: { title: journal['dc:title'] || '', issn: journal['prism:issn'] || '', publisher: journal['dc:publisher'] || '', coverage: `${journal['coverageStartYear'] || ''} - ${journal['coverageEndYear'] || 'حالياً'}`, citeScore: citeScore, url: `https://www.scopus.com/source/sourceInfo.url?sourceId=${journal['source-id'] || ''}` } }; return { statusCode: 200, headers, body: JSON.stringify(result) }; } catch (error) { return { statusCode: 500, headers, body: JSON.stringify({ error: error.message }) }; } };
